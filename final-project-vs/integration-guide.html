📖 Database Integration Guide

Learn how to add user authentication and data storage to your existing pages.

Step 1: Include the Database Client

Add this line to the <head> section of your HTML file:

<script src="db-client.js"></script>

This gives you access to the global db object.

Step 2: Add a Login Section

Create a login form in your HTML:

<div id="loginForm">
    <input type="text" id="username" placeholder="Username">
    <input type="password" id="password" placeholder="Password">
    <button onclick="login()">Login</button>
</div>

<div id="userGreeting" style="display:none;">
    Welcome, <span id="userName"></span>!
    <button onclick="logout()">Logout</button>
</div>

Step 3: Add JavaScript Functions

<script>
// Check if user is logged in on page load
window.onload = () => {
    if (db.isLoggedIn()) {
        showUserGreeting();
    }
};

// Show user greeting
function showUserGreeting() {
    document.getElementById('loginForm').style.display = 'none';
    document.getElementById('userGreeting').style.display = 'block';
    document.getElementById('userName').textContent = db.username;
}

// Login function
async function login() {
    const username = document.getElementById('username').value;
    const password = document.getElementById('password').value;
    
    const result = await db.login(username, password);
    if (result.userId) {
        showUserGreeting();
    } else {
        alert('Login failed: ' + result.error);
    }
}

// Logout function
function logout() {
    db.logout();
    document.getElementById('loginForm').style.display = 'block';
    document.getElementById('userGreeting').style.display = 'none';
    document.getElementById('username').value = '';
    document.getElementById('password').value = '';
}
</script>

Step 4: Save User-Specific Data

Store data tied to the logged-in user:

<button onclick="saveFavoriteSong()">Save Favorite Song</button>

<script>
async function saveFavoriteSong() {
    if (!db.isLoggedIn()) {
        alert('Please login first');
        return;
    }
    
    const song = prompt('Enter your favorite song:');
    if (song) {
        await db.saveData('favorite_song', song);
        alert('Saved!');
    }
}

// Load user data
async function loadUserData() {
    if (!db.isLoggedIn()) return;
    
    const favSong = await db.getData('favorite_song');
    if (favSong) {
        console.log('Your favorite song:', favSong.data_value);
    }
}
</script>

Common Tasks Reference

Check if User is Logged In

if (db.isLoggedIn()) {
    console.log('User:', db.username);
} else {
    console.log('Not logged in');
}

Save Different Types of Data

// Save simple text
await db.saveData('bio', 'I love music');

// Save a number
await db.saveData('high_score', '9500');

// Save JSON object
await db.saveObject('preferences', {
    theme: 'dark',
    notifications: true
});

// Save array
await db.saveObject('favorite_songs', [
    'Song 1',
    'Song 2',
    'Song 3'
]);

Retrieve Data

// Get specific data
const bio = await db.getData('bio');
console.log(bio.data_value);

// Get parsed object
const prefs = await db.getObject('preferences');
console.log(prefs.theme);

// Get all user data
const allData = await db.getAllData();
allData.forEach(item => {
    console.log(item.data_key + ': ' + item.data_value);
});

Delete Data

await db.deleteData('high_score');

Handle Errors

async function loginSafely() {
    try {
        const result = await db.login(username, password);
        if (!result.userId) {
            console.error('Login failed:', result.error);
        }
    } catch (error) {
        console.error('Error:', error);
    }
}

Integration Examples by Page

Leaderboard Page

Add user-specific scores to the leaderboard:

// Save user's score
async function submitScore(score) {
    if (!db.isLoggedIn()) {
        alert('Please login to submit a score');
        return;
    }
    await db.saveData('leaderboard_score', score);
    await db.saveData('last_submitted', new Date().toLocaleString());
}

// Load user's score
async function loadUserScore() {
    const scoreData = await db.getData('leaderboard_score');
    if (scoreData) {
        document.getElementById('userScore').textContent = scoreData.data_value;
    }
}

Import Page

Save import history per user:

async function saveImportHistory(songCount) {
    if (!db.isLoggedIn()) return;
    
    await db.saveObject('last_import', {
        count: songCount,
        date: new Date().toLocaleString()
    });
}

Info Page

Display user profile information:

async function displayUserInfo() {
    if (!db.isLoggedIn()) return;
    
    const profile = await db.getUserProfile();
    const allData = await db.getAllData();
    
    console.log('Profile:', profile);
    console.log('Saved Data:', allData);
}

💡 Tip: The db object automatically saves the user ID and username to browser storage. This persists even after the user closes the browser, so they'll stay logged in!
⚠️ Remember: Always check db.isLoggedIn() before trying to save or retrieve user data.
